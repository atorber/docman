## 创建服务
使用以下代码可以创建在线部署任务

```go
// import api "github.com/baidubce/bce-sdk-go/services/aihc/inference/v2"
ak, sk, endpoint := "Your ak", "Your sk", "aihc.bj.baidubce.com"
client, _ := api.NewClient(ak, sk, endpoint)
createServiceArgs ：= &api.ServiceConf{
    Name:            "test",
    InstanceCount:   1,
    AcceleratorType: "",
    WorkloadType:    "",
    Containers: []api.ContainerConf{
        {
            Name:             "test-container",
            Cpus:             1,
            Memory:           1,
            AcceleratorCount: 0,
            Command:          []string{"/bin/sh", "-c", "sleep inf"},
            Image: api.ImageConf{
                ImageType: 1,
                ImageUrl:  "your image url",
            },
            Ports: []api.PortConf{
                {
                    Name: "HTTP",
                    Port: 8080,
                },
            },
        },
    },
    ResourcePool: api.ResourcePoolConf{
        ResourcePoolId:   "your resource pool id",
        ResourcePoolName: "your resource pool name",
        QueueName:        "your queue name",
    },
    Misc: api.Misc{
        GracePeriodSec: 30,
        FedPodsPerIns:  0,
    },
    Deploy: api.DeployConf{
        Schedule: api.ScheduleConf{
            Priority: "high",
        },
    },
    Access: api.AccessConf{
        PublicAccess: false,
        NetworkType:  "aiGateway",
        AiGateway: api.AiGatewayConf{
            EnableAuth: true,
        },
    },
}

result, err := client.CreateService(createServiceArgs, "your clientToken")

if err != nil {
    panic(err)
}

jsonBytes, _ := json.Marshal(result)
fmt.Println(string(jsonBytes))

```
注意:

* 根据接口文档填写具体的访问参数
